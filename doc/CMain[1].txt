function isValidDate(dateStr,format)
{
	var yearPos = format.indexOf("YYYY");
	var monthPos = format.indexOf("MM");
	var dayPos = format.indexOf("DD");
	
	if(dateStr.length != format.length)
	{
		return false;
	}
	
	var year = dateStr.substring(yearPos,yearPos + 4);
	var month = dateStr.substring(monthPos,monthPos + 2);
	var day = dateStr.substring(dayPos,dayPos + 2); 

	if(isNaN(year) || isNaN(month) || isNaN(day)){
		return false;
	}
	
	var dayNumOfMonth = new Array(31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31);
	
	if(((year % 4 == 0) && (year % 100 != 0)) || (year % 400 == 0)){
		dayNumOfMonth[1]=29;
	}
	
	if(month < 1 || month > 12 || day >dayNumOfMonth[month -1] || day < 1 ){
		return false;
	}
	else{
		return true;
	}
}

	var objRate = window.document.getElementById('CCAS_PLAATRN.LOANRATE');
	if (objRate == null) {
		return;
	}

	if (Trim(window.document.getElementById('CCAS_PLAATRN.LAFDRAMT').value)!='') {
		jkamt = window.document.getElementById('CCAS_PLAATRN.LAFDRAMT').value;
	}
	jkamt = jkamt*1.0 + oldamt*1.0;
	var totamt = 0;
	if (Trim(window.document.getElementById('CCAS_RETAILTRAN.PRICE').value)!='') {
		totamt = window.document.getElementById('CCAS_RETAILTRAN.PRICE').value;
	}
	if (procd == "PLAA" || procd == "PLAB" || procd == "PLAC" || procd == "PLAD" || procd == "PLAF" || procd == "PLCB" || procd == "PLHC" ) {
		var tmpamt = 0;
		if (Trim(window.document.getElementById('CCAS_RETAILTRAN.TOTPRICE').value)!='') {
			tmpamt = window.document.getElementById('CCAS_RETAILTRAN.TOTPRICE').value;
		}
		if (tmpamt*1.0 > 0 && totamt*1.0 > tmpamt*1.0) {
			totamt = tmpamt;
		}
	} else if (procd == "PLBA" || procd == "PLBB" || procd == "PLBC" || procd == "PLBD" ) {
		if (Trim(window.document.getElementById('CCAS_PLAATRN.GUMOD').value)=='D5') {
			var tmpamt = 0;
			if (Trim(window.document.getElementById('CCAS_CCTCTI.PRICE_1').value)!='') {
				tmpamt = window.document.getElementById('CCAS_CCTCTI.PRICE_1').value;
			}
			if (tmpamt*1.0 > 0 && totamt*1.0 > tmpamt*1.0) {
				totamt = tmpamt;
			}
		}
	}
	var rate = 0.00;
	if (totamt != 0) {
		rate = adv_format(jkamt/totamt*100,2);
	}
	objRate.value = rate;
}

function validateMoneyLength(formObj) {
       if (formObj.value.length==0) {return true;}

	formObj.value = Trim(formObj.value);
	var objvalue = Trim(formObj.value);
	if(objvalue.length > 0){
	   if(objvalue.substr(0,1)=='-')
	      objvalue= objvalue.substr(1,objvalue.length);	
	}
	var maxLength = (formObj.maxLength)*1.0;
	var moneySyntax;
	// No params set, all special chars become optional
	moneyReg="^[0-9]{1," + (maxLength-2) + "}(\\.[0-9]{1,2})?$"; moneySyntax="XXXXXXX("+ (maxLength-2)+"位)[.XX](2位)";
	var regex = new RegExp(moneyReg);
	if (!regex.test(objvalue)) {
	   return errorProcess3(formObj,1,1,formObj.value+'请按这个格式进行输入 '+moneySyntax+' for '+objName+'.');
	}
	return true;
}

function caculate21(ll,qs,je)
{
   return ((ll * Math.pow((1+ll),qs)) / (Math.pow(1+ll,qs)-1) * je);   
}

function caculate22( ll, qs, je, bj)
{
        //每月供款额=贷款金额/供款总期数+（贷款金额－已还本金）×月利率
        return (je/qs+(je-bj)*ll);
}
